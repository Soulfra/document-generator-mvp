#!/bin/bash
# deploy-subdomain.sh

SUBDOMAIN=$1

VERSION=${3:-latest}

echo "üöÄ Deploying $SERVICE to $SUBDOMAIN..."

# Update service
docker pull "gaming-platform/${SERVICE}:${VERSION}"

# Rolling update

  --image "gaming-platform/${SERVICE}:${VERSION}" \
  --update-parallelism 1 \
  --update-delay 30s \
  "${SERVICE}"

# Verify health
sleep 10
curl -f "https://${SUBDOMAIN}/health" || exit 1

echo "‚úÖ $SUBDOMAIN deployment complete"
